## 문제 링크
  [26068번 - 치킨댄스를 추는 곰곰이를 본 임스](https://www.acmicpc.net/problem/26068)

## 설명
  문자열에서 정수를 파싱하는 것과 조건문을 이용한 구현 문제입니다.


- - -

## Code
  ```c#
namespace Solution {
  class Program {
    static void Main(string[] args) {

      int.TryParse(Console.ReadLine(), out int n);

      int ans = 0;
      for (int i = 0; i < n; i++) {
        string? input = Console.ReadLine();
        int.TryParse(input!.Substring(2), out int dayLeft);
        if (dayLeft <= 90) ans++;
      }

      Console.WriteLine(ans);

    }
  }
}
  ```

#include <bits/stdc++.h>

using namespace std;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n; cin >> n;

  int ans = 0;
  for (int i = 0; i < n; i++) {
    string giftycon; cin >> giftycon;
    int dayLeft = stoi(giftycon.substr(2));
    if (dayLeft <= 90) ans++;
  }

  cout << ans << "\n";

  return 0;
}
