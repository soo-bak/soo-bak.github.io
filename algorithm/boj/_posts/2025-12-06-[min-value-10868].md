## 문제 링크
[10868번 - 최솟값](https://www.acmicpc.net/problem/10868)

## 설명

N개의 정수로 이루어진 수열에서 구간 최솟값을 M번 질의하는 문제입니다.

<br>

## 접근법

N과 M이 각각 최대 10만이므로 매 질의마다 구간을 순회하면 시간 초과가 발생합니다. 세그먼트 트리를 사용하면 각 질의를 O(log N)에 처리할 수 있습니다.

세그먼트 트리의 각 노드는 해당 구간의 최솟값을 저장합니다. 리프 노드에 원본 배열 값을 넣고, 부모 노드는 두 자식 중 작은 값을 가집니다. 질의 시에는 구간이 완전히 포함되면 해당 노드 값을 반환하고, 일부만 겹치면 자식으로 내려가 결과를 합칩니다.

<br>

## Code

### C#

```csharp
using System;
using System.IO;

namespace Solution {
  class Program {
    static int[] tree = Array.Empty<int>();
    static int n, size;

    static void Build(int[] arr) {
      size = 1;
      while (size < n)
        size <<= 1;
      tree = new int[size << 1];
      Array.Fill(tree, int.MaxValue);
      for (var i = 0; i < n; i++)
        tree[size + i] = arr[i];
      for (var i = size - 1; i > 0; i--)
        tree[i] = Math.Min(tree[i << 1], tree[i << 1 | 1]);
    }

    static int Query(int l, int r) {
      l += size;
      r += size;
      var ret = int.MaxValue;
      while (l <= r) {
        if ((l & 1) == 1)
          ret = Math.Min(ret, tree[l++]);
        if ((r & 1) == 0)
          ret = Math.Min(ret, tree[r--]);
        l >>= 1;
        r >>= 1;
      }
      return ret;
    }

    static void Main(string[] args) {
      using var sr = new StreamReader(Console.OpenStandardInput());
      using var sw = new StreamWriter(Console.OpenStandardOutput());

      var first = sr.ReadLine()!.Split();
      n = int.Parse(first[0]);
      var m = int.Parse(first[1]);

      var arr = new int[n];
      for (var i = 0; i < n; i++)
        arr[i] = int.Parse(sr.ReadLine()!);

      Build(arr);

      for (var i = 0; i < m; i++) {
        var parts = sr.ReadLine()!.Split();
        var a = int.Parse(parts[0]) - 1;
        var b = int.Parse(parts[1]) - 1;
        sw.WriteLine(Query(a, b));
      }
    }
  }
}
```

### C++

```cpp
#include <bits/stdc++.h>
using namespace std;

typedef vector<int> vi;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int N, M; cin >> N >> M;
  int size = 1;
  while (size < N)
    size <<= 1;

  const int INF = 1'000'000'001;
  vi seg(size << 1, INF);

  for (int i = 0; i < N; i++)
    cin >> seg[size + i];
  for (int i = size - 1; i > 0; i--)
    seg[i] = min(seg[i << 1], seg[i << 1 | 1]);

  while (M--) {
    int l, r; cin >> l >> r;
    l = l - 1 + size;
    r = r - 1 + size;
    int ret = INF;
    while (l <= r) {
      if (l & 1)
        ret = min(ret, seg[l++]);
      if (!(r & 1))
        ret = min(ret, seg[r--]);
      l >>= 1;
      r >>= 1;
    }
    cout << ret << "\n";
  }

  return 0;
}
```
