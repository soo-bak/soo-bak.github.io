## 문제 링크
[7168번 - Nonogramm](https://www.acmicpc.net/problem/7168)

## 설명
격자가 주어질 때, 행마다 그리고 열마다 연속된 검은칸 블록들의 길이를 순서대로 출력하는 문제입니다.

블록이 없으면 0을 출력합니다.

<br>

## 접근법
먼저 각 행을 순회하며 연속된 검은칸의 길이를 세어 리스트에 담습니다.

다음으로 각 열에 대해서도 동일한 방식으로 처리합니다.

이후 리스트가 비었으면 0을, 아니면 길이들을 순서대로 출력합니다.

- - -

## Code

### C#
```csharp
using System;
using System.Collections.Generic;
using System.Text;

class Program {
  static void Main() {
    var first = Console.ReadLine()!.Split();
    var n = int.Parse(first[0]);
    var m = int.Parse(first[1]);
    var grid = new string[n];
    for (var i = 0; i < n; i++)
      grid[i] = Console.ReadLine()!;

    var sb = new StringBuilder();

    for (var i = 0; i < n; i++) {
      var seg = new List<int>();
      var cnt = 0;
      for (var j = 0; j < m; j++) {
        if (grid[i][j] == '#') cnt++;
        else {
          if (cnt > 0) seg.Add(cnt);
          cnt = 0;
        }
      }
      if (cnt > 0) seg.Add(cnt);

      if (seg.Count == 0) sb.AppendLine("0");
      else sb.AppendLine(string.Join(" ", seg));
    }

    for (var j = 0; j < m; j++) {
      var seg = new List<int>();
      var cnt = 0;
      for (var i = 0; i < n; i++) {
        if (grid[i][j] == '#') cnt++;
        else {
          if (cnt > 0) seg.Add(cnt);
          cnt = 0;
        }
      }
      if (cnt > 0) seg.Add(cnt);

      if (seg.Count == 0) sb.AppendLine("0");
      else sb.AppendLine(string.Join(" ", seg));
    }

    Console.Write(sb.ToString());
  }
}
```

#include <bits/stdc++.h>
using namespace std;

typedef vector<int> vi;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n, m; cin >> n >> m;
  vector<string> grid(n);
  for (int i = 0; i < n; i++)
    cin >> grid[i];

  for (int i = 0; i < n; i++) {
    vi seg;
    int cnt = 0;
    for (int j = 0; j < m; j++) {
      if (grid[i][j] == '#') cnt++;
      else {
        if (cnt > 0) seg.push_back(cnt);
        cnt = 0;
      }
    }
    if (cnt > 0) seg.push_back(cnt);

    if (seg.empty()) cout << "0\n";
    else {
      for (size_t k = 0; k < seg.size(); k++) {
        if (k) cout << " ";
        cout << seg[k];
      }
      cout << "\n";
    }
  }

  for (int j = 0; j < m; j++) {
    vi seg;
    int cnt = 0;
    for (int i = 0; i < n; i++) {
      if (grid[i][j] == '#') cnt++;
      else {
        if (cnt > 0) seg.push_back(cnt);
        cnt = 0;
      }
    }
    if (cnt > 0) seg.push_back(cnt);

    if (seg.empty()) cout << "0\n";
    else {
      for (size_t k = 0; k < seg.size(); k++) {
        if (k) cout << " ";
        cout << seg[k];
      }
      cout << "\n";
    }
  }

  return 0;
}
