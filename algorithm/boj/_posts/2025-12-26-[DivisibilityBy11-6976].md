## 문제 링크
[6976번 - Divisibility by 11](https://www.acmicpc.net/problem/6976)

## 설명
주어진 수에 대해 규칙을 반복 적용하며 생성된 수들을 출력하고 11의 배수 여부를 판별하는 문제입니다.

<br>

## 접근법
먼저 현재 수를 출력하고, 길이가 두 자리 이하가 될 때까지 규칙을 반복합니다.

다음으로 마지막 자리를 제거하고 그 값을 줄어든 수에서 빼서 새로운 수를 만듭니다.

이후 과정에서 생기는 앞자리 0은 제거해 유효한 수로 유지합니다.

마지막으로 남은 수가 11의 배수인지 판단해 문장을 출력합니다.

<br>

## Code

### C#
```csharp
using System;
using System.Text;

class Program {
  static string SubtractDigit(string s, int d) {
    if (d == 0) {
      var pos0 = 0;
      while (pos0 < s.Length - 1 && s[pos0] == '0') pos0++;
      return s.Substring(pos0);
    }

    var arr = s.ToCharArray();
    var i = arr.Length - 1;
    var borrow = d;
    while (borrow > 0 && i >= 0) {
      var val = arr[i] - '0';
      if (val >= borrow) {
        val -= borrow;
        arr[i] = (char)('0' + val);
        borrow = 0;
      } else {
        val = val + 10 - borrow;
        arr[i] = (char)('0' + val);
        borrow = 1;
      }
      i--;
    }

    var pos = 0;
    while (pos < arr.Length - 1 && arr[pos] == '0') pos++;
    return new string(arr, pos, arr.Length - pos);
  }

  static void Main() {
    var parts = Console.In.ReadToEnd().Split();
    var idx = 0;
    var t = int.Parse(parts[idx++]);
    var sb = new StringBuilder();

    for (var caseNum = 0; caseNum < t; caseNum++) {
      var original = parts[idx++];
      var s = original;

      var outSb = new StringBuilder();
      outSb.AppendLine(s);
      while (s.Length > 2) {
        var d = s[s.Length - 1] - '0';
        s = SubtractDigit(s.Substring(0, s.Length - 1), d);
        outSb.AppendLine(s);
      }

      var val = int.Parse(s);
      if (val % 11 == 0) outSb.Append("The number ").Append(original).Append(" is divisible by 11.");
      else outSb.Append("The number ").Append(original).Append(" is not divisible by 11.");

      sb.Append(outSb);
      if (caseNum != t - 1) sb.AppendLine().AppendLine();
    }

    Console.Write(sb);
  }
}
```

### C++
```cpp
#include <bits/stdc++.h>
using namespace std;

string subtractDigit(const string &s, int d) {
  if (d == 0) {
    int pos0 = 0;
    while (pos0 < (int)s.size() - 1 && s[pos0] == '0') pos0++;
    return s.substr(pos0);
  }

  string res = s;
  int i = (int)res.size() - 1;
  int borrow = d;
  while (borrow > 0 && i >= 0) {
    int val = res[i] - '0';
    if (val >= borrow) {
      val -= borrow;
      res[i] = char('0' + val);
      borrow = 0;
    } else {
      val = val + 10 - borrow;
      res[i] = char('0' + val);
      borrow = 1;
    }
    i--;
  }

  int pos = 0;
  while (pos < (int)res.size() - 1 && res[pos] == '0') pos++;
  return res.substr(pos);
}

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int t; cin >> t;
  for (int caseNum = 0; caseNum < t; caseNum++) {
    string original; cin >> original;
    string s = original;

    cout << s << "\n";
    while ((int)s.size() > 2) {
      int d = s.back() - '0';
      s = subtractDigit(s.substr(0, s.size() - 1), d);
      cout << s << "\n";
    }

    int val = stoi(s);
    if (val % 11 == 0) cout << "The number " << original << " is divisible by 11.\n";
    else cout << "The number " << original << " is not divisible by 11.\n";

    if (caseNum != t - 1) cout << "\n";
  }

  return 0;
}
```
