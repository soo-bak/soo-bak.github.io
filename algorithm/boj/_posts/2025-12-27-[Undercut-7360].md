## 문제 링크
[7360번 - Undercut](https://www.acmicpc.net/problem/7360)

## 설명
두 플레이어가 n라운드 동안 낸 카드가 주어질 때, 라운드별 규칙에 따라 점수를 누적해 최종 합을 출력하는 문제입니다.

<br>

## 접근법
각 라운드에서 두 카드가 같으면 아무도 점수를 얻지 못합니다.

두 카드의 차이가 1이면 작은 카드를 낸 쪽이 두 카드의 합만큼 점수를 얻습니다. 단, 1과 2 조합은 특별히 6점입니다.

그 외의 경우에는 큰 카드를 낸 쪽이 자신의 카드 값만큼 점수를 얻습니다.

## Code

### C#
```csharp
using System;
using System.Text;

class Program {
  static void Main() {
    var parts = Console.In.ReadToEnd().Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
    var idx = 0;
    var sb = new StringBuilder();
    var first = true;

    while (idx < parts.Length) {
      var n = int.Parse(parts[idx++]);
      if (n == 0) break;

      var a = new int[n];
      var b = new int[n];
      for (var i = 0; i < n; i++) a[i] = int.Parse(parts[idx++]);
      for (var i = 0; i < n; i++) b[i] = int.Parse(parts[idx++]);

      int scoreA = 0, scoreB = 0;
      for (var i = 0; i < n; i++) {
        var x = a[i];
        var y = b[i];
        if (x == y) continue;

        var diff = Math.Abs(x - y);
        if (diff == 1) {
          var low = Math.Min(x, y);
          var high = Math.Max(x, y);
          var gain = (low == 1 && high == 2) ? 6 : x + y;
          if (x < y) scoreA += gain;
          else scoreB += gain;
        } else {
          if (x > y) scoreA += x;
          else scoreB += y;
        }
      }

      if (!first) sb.AppendLine();
      first = false;
      sb.AppendLine($"A has {scoreA} points. B has {scoreB} points.");
    }

    Console.Write(sb);
  }
}
```

### C++

#include <bits/stdc++.h>
using namespace std;

typedef vector<int> vi;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n;
  bool first = true;
  while (cin >> n) {
    if (n == 0) break;
    vi a(n), b(n);
    for (int i = 0; i < n; i++) cin >> a[i];
    for (int i = 0; i < n; i++) cin >> b[i];

    int scoreA = 0, scoreB = 0;
    for (int i = 0; i < n; i++) {
      int x = a[i], y = b[i];
      if (x == y) continue;
      int diff = abs(x - y);

      if (diff == 1) {
        int low = min(x, y), high = max(x, y);
        int gain = (low == 1 && high == 2) ? 6 : x + y;
        if (x < y) scoreA += gain;
        else scoreB += gain;
      } else {
        if (x > y) scoreA += x;
        else scoreB += y;
      }
    }

    if (!first) cout << "\n";
    first = false;
    cout << "A has " << scoreA << " points. B has " << scoreB << " points." << "\n";
  }

  return 0;
}
